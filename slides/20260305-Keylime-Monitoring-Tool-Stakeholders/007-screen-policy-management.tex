\begin{frame}{Policy Management -- Eliminate Blind Changes}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{alertblock}{Before: Policy Changes Are Risky}
        \scriptsize
        \begin{itemize}
          \item No way to preview impact
          \item Changes are immediate and global
          \item Mistakes affect entire fleet
          \item No rollback capability
          \item No audit trail of changes
        \end{itemize}
      \end{alertblock}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{exampleblock}{After: Safe, Controlled Changes}
        \scriptsize
        \begin{itemize}
          \item \textbf{Impact analysis} before applying
          \item \textbf{Staged rollout} to test groups first
          \item \textbf{Two-person approval} for critical policies
          \item \textbf{One-click rollback} if issues arise
          \item \textbf{Full version history} with diffs
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
