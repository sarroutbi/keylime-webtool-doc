\begin{frame}{Audit Trail -- Built for Compliance}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Every Action Is Recorded}
        \scriptsize
        The audit log captures \textbf{who} did \textbf{what},
        \textbf{when}, and \textbf{why} -- across the entire
        Keylime infrastructure.
      \end{block}
      \scriptsize
      \begin{itemize}
        \item Agent registration and lifecycle events
        \item Policy changes with before/after diffs
        \item Administrative actions with user identity
        \item Attestation failures and resolutions
        \item Certificate operations
        \item Login and access events
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Tamper-Evident by Design}
        \scriptsize
        Logs are \textbf{hash-chained} -- any modification
        or deletion is immediately detectable.
      \end{block}
      \scriptsize
      \begin{itemize}
        \item Each entry includes hash of previous entry
        \item Periodic checkpoints for verification
        \item Exportable for external auditing
        \item Meets SOC 2 and FedRAMP requirements
      \end{itemize}
