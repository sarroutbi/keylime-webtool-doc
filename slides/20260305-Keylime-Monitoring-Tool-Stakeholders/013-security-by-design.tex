\section{Security by Design}

\begin{frame}{Security -- Built In, Not Bolted On}
  \begin{block}{A Security Tool Must Be Secure Itself}
    \scriptsize
    The dashboard holds privileged access to the entire Keylime infrastructure.
    Security is not a Phase 3 feature -- it is the \textbf{foundation from Day 1}.
  \end{block}

  \vspace{0.1cm}

  \begin{center}
    \begin{tikzpicture}[scale=0.6, transform shape,
        layer/.style={rectangle, draw, rounded corners, text width=10cm,
                      text centered, minimum height=0.9cm, font=\scriptsize, inner sep=0.15cm}]

      \node[layer, fill=dangered!8] (l4) at (0,3.6) {
        \faLock~\textbf{Authentication \& Authorization}\\
        \tiny OIDC/SAML login with MFA | Three-tier RBAC (Viewer / Operator / Admin)
      };

      \node[layer, fill=warningyellow!8] (l3) at (0,2.2) {
        \faIcon{shield-alt}~\textbf{Encrypted Communications}\\
        \tiny TLS 1.3 for users | mTLS for Keylime API access | All hops encrypted
      };

      \node[layer, fill=keylimeblue!8] (l2) at (0,0.8) {
        \faKey~\textbf{Secret Protection}\\
        \tiny Private keys in HSM/Vault only, never on disk | Short-lived session tokens
      };

      \node[layer, fill=successgreen!8] (l1) at (0,-0.6) {
        \faIcon{clipboard-check}~\textbf{Tamper-Evident Audit Trail}\\
        \tiny Hash-chained logs | Every action recorded with user identity | Compliance-ready
      };
    \end{tikzpicture}
  \end{center}

  \vspace{0.1cm}

  \begin{alertblock}{Defense in Depth}
    \scriptsize
    Even if one layer is compromised, the remaining layers limit the blast radius.
    Two-person approval for policy changes prevents a single compromised admin session
    from pushing malicious configurations.
  \end{alertblock}
\end{frame}

\begin{frame}{Role-Based Access Control}
  \begin{block}{Right Access for the Right People}
    \scriptsize
    Three roles ensure each team member sees only what they need and can do only
    what they're authorized to do.
  \end{block}

  \vspace{0.2cm}

  \begin{center}
