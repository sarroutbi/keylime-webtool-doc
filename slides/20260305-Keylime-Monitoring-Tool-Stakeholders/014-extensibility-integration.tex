\begin{frame}{Built to Scale, Built to Extend}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{block}{Scalability}
        \tiny
        Designed to grow with your infrastructure.
      \end{block}
      \tiny
      \begin{itemize}
        \item Supports \textbf{both pull and push} attestation modes
        \item Scales from 10 to \textbf{100,000+ agents}
        \item High availability deployment options
        \item Multi-datacenter and multi-cluster support
        \item Event-driven architecture minimizes load
      \end{itemize}

      \vspace{0.1cm}

      \begin{block}{Deployment Flexibility}
        \tiny
        Runs anywhere your infrastructure runs.
      \end{block}
      \tiny
      \begin{itemize}
        \item Container images (Podman/Docker)
        \item Kubernetes via Helm chart
        \item RPM package for bare metal
        \item \textbf{Air-gapped environments} supported
        \item Self-contained, no CDN dependencies
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{block}{Extensibility}
        \tiny
        Open architecture that welcomes additions.
      \end{block}
      \tiny
      \begin{itemize}
        \item \textbf{API-first} design -- all features accessible via REST
        \item Non-invasive: \textbf{zero changes} to Keylime itself
        \item Configurable alert thresholds and rules
        \item Custom compliance framework definitions
        \item Webhook integrations for any service
        \item Multi-tenancy for shared environments
      \end{itemize}

      \vspace{0.1cm}

      \begin{exampleblock}{Future-Proof}
        \tiny
        \begin{itemize}
          \item Supports Keylime API v2 and v3
          \item Graceful handling of API version changes
          \item Mobile app path via shared React codebase
          \item WCAG 2.1 AA accessibility compliance
        \end{itemize}
      \end{exampleblock}
    \end{column}
  \end{columns}
\end{frame}
